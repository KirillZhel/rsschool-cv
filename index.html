    double lagrange1(double* x, double* y, int n, double _x)
    {
        double result = 0.0;

        for (int i = 0; i < n; i++)
        {
            double P = 1.0;

            for (int j = 0; j < n; j++)
                if (j != i)
                    P *= (_x - x[j]) / (x[i] - x[j]);

            result += P * y[i];
        }

        return result;
    }
